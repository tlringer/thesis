(* User A *)
Fixpoint bin_to_nat (b: bin) :=
  match b with
  | B0 => O
  | B2 b' =>
      2 * (bin_to_nat b')
  | B21 b' =>
      1 + 2 * (bin_to_nat b')
  end.

(* User B *)
Fixpoint bin_to_nat (b: bin) :=
  match b with
  | B0 => O
  | B2 b' =>
      (bin_to_nat b') +
      (bin_to_nat b')
  | B21 b' =>
      S ((bin_to_nat b') +
         (bin_to_nat b'))
  end.
